%
% \begin{Display}{p}{Label}{This is the Caption} 
% ...
% \end{Display}
%
% This creates a boxed figure in a uniform style.
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newenvironment{Display}[3]{%
\begin{figure}[#1]%
\leftskip=\baseLeftSkip%
\label{#2}
\vspace{1mm}%
%\caption{#3}
\def\tmpdisplayname{#3}
\begin{tabular}{|c|}%
\hline \hline
\begin{small}%
\begin{minipage}[t]{\xmpBoxWidth}%
}{%
\end{minipage}%
\end{small}%
\\
\\ \hline\hline
\end{tabular}%
\caption{\tmpdisplayname}
\end{figure}
}
\newenvironment{UnboxedWideDisplay}[3]{%
\begin{figure}[#1]%
\leftskip=0pt
\def\tmpdisplayname{#3}
\label{#2}
\vspace{1mm}%
\begin{tabular*}{\textwidth}{c}%
\hline \hline
\begin{small}%
\begin{minipage}[t]{\textwidth}%
}{%
\end{minipage}%
\end{small}%
\\
\end{tabular*}%
\caption{\tmpdisplayname}
\end{figure}
}
